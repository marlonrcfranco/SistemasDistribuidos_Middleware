# - Hello.java
# - HelloServer.java
#
# COMPILA CODIGOS
javac *.java
#
# EXECUTA RMI DA CLASSE QUE IMPLEMENTA A INTERFACE
rmic Hello
#
# EXECUTA REGISTRY
rmiregistry
#
# EXECUTA CODIGO DO SERVIDOR
java HelloServer.java


# ******* [CLIENTE] *******
#
# Tem os codigos:
# - HelloInterface.java
# - Hello_Stub.class (gerado pelo rmic Hello no Servidor e copiado e colado aqui)
# - HelloClient.java
#
# COMPILA CODIGOS
javac *.java
#
# EXECUTA CODIGO DO SERVIDOR
java HelloClient.java

# Ver processos java em execução
ps -fC java

# Seguranca (slides da aula)
i) java -Djava.security.policy=policy.all HelloClient
ii) java -Djava.security.policy=server.policy HelloClient
#
# Para segundo exemplo, a política de segurança é informada em um arquivo “server.policy”, com o seguinte conteúdo:
#
grant {
    permission java.security.AllPermission;
};
